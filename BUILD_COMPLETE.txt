================================================================================
                        DATABASE MODELS (4)
================================================================================

USER MODEL
  email (unique) - string
  name - string
  password - hashed string
  role - enum: student | company | coordinator
  skills - array (for students)
  companyName - string (for companies)
  companyId - string (for data isolation)
  resume - string (optional)
  createdAt, updatedAt - timestamps

JOB MODEL
  title - string
  description - string
  company - ObjectId (reference to User)
  companyName - string
  companyId - string (for isolation)
  requiredSkills - array
  applications - array of ObjectIds
  isOpen - boolean
  deadline - Date (optional)
  createdAt, updatedAt - timestamps

APPLICATION MODEL
  student - ObjectId (reference to User)
  studentName - string
  studentEmail - string
  job - ObjectId (reference to Job)
  company - ObjectId (reference to User)
  companyName - string
  status - enum: pending | accepted | rejected
  matchScore - number (0-100)
  studentSkills - array
  createdAt, updatedAt - timestamps

COORDINATORSETTINGS MODEL
  isFrozen - boolean
  frozenAt - Date (optional)
  frozenBy - ObjectId (reference to User, optional)
  reason - string (optional)
  createdAt, updatedAt - timestamps
